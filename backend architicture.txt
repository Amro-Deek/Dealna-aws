/ (Root Project)
├── 📂 database/                      # Infrastructure Source of Truth (Database Blueprint)
│   ├── 📂 migrations/                # SQL Migration files (Up/Down) with timestamps
│   ├── 📂 query/                     # Raw SQL files for sqlc (user.sql, activation.sql, etc.)
│   └── 📄 schema.sql                 # Global DDL (Table definitions) read by sqlc
│
└── 📂 backend/                       # Go Application Core
    ├── 📄 .env                       # Environment variables (DB_URL, JWT_SECRET)
    ├── 📄 sqlc.yaml                  # Config mapping /database to /backend/internal/database/generated
    ├── 📄 Makefile                   # Automation (make sqlc, make migrate-create)
    ├── 📄 go.mod                     # Go dependencies
    ├── 📂 cmd/server/
    │   └── 📄 main.go                # Entry Point: Config loading, DB init, & DI Bootstrap
    │
    ├── 📂 internal/
    │   ├── 📂 core/                  # "The Hexagon Heart" (Business Logic) - NO external deps
    │   │   ├── 📂 domain/            # Business Entities (user.go) - Pure Go structs
    │   │   ├── 📂 ports/             # Interfaces only (user_repository.go, auth.go)
    │   │   └── 📂 services/          # Use-cases (auth_service.go) - Orchestrates business rules
    │   │
    │   ├── 📂 adapters/              # "The Adapters" (Interface Implementation)
    │   │   ├── 📂 primary/           # (Incoming) - Triggered by external actors (e.g., HTTP)
    │   │   │   ├── 📂 auth/          # Auth Adapter
    │   │   │   │   ├── 📂 dto/       # API Contracts (login.go) - JSON tags only
    │   │   │   │   ├── 📂 http/      # Auth-specific HTTP Handlers and local routes
    │   │   │   │   └── 📄 handler.go # Platform-agnostic logic (Bind DTO -> Call Service)
    │   │   │   ├── 📂 users/         # Users Adapter (dto, http, handler.go)
    │   │   │   └── 📄 routes.go      # ⭐ App-level Router: Aggregates all handlers via chi router
    │   │   │
    │   │   └── 📂 secondary/         # (Outgoing) - Called by the application
    │   │       ├── 📂 persistence/   # Database Implementations
    │   │       │   └── 📂 postgres/  # Uses sqlc-generated code to fulfill Port interfaces
    │   │       │       ├── 📄 user_repo.go
    │   │       │       └── 📄 factory.go    # Repository Factory pattern
    │   │       └── 📂 auth/          # Auth Provider implementations (jwt_provider.go)
    │   │
    │   ├── 📂 database/              # Low-level Database Connectivity
    │   │   ├── 📂 generated/         # Auto-generated by sqlc (db.go, models.go) - DO NOT EDIT
    │   │   ├── 📄 connect.go         # SQL connection pooling (sql.Open)
    │   │   └── 📄 db.go              # DB initialization and Transaction handling logic
    │   │
    │   ├── 📂 middleware/            # Auth, Logging, Errors, Recovery, Context
    │   └── 📂 config/                # Environment config mapping
    │
    └── 📂 docs/                      # API Documentation (Swagger JSON/YAML)
